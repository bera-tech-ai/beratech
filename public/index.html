// Welcome to Real-time Code Collaboration!
// Multiple developers can edit this code simultaneously

class CollaborationSession {
    constructor() {
        this.features = [
            "Real-time code editing",
            "Live cursor positions", 
            "Multi-user collaboration",
            "Syntax highlighting",
            "Auto-save functionality"
        ];
        this.participants = new Map();
    }

    addParticipant(user) {
        this.participants.set(user.id, user);
        this.updateCollaboratorsUI();
    }

    updateCollaboratorsUI() {
        const count = this.participants.size;
        document.getElementById('collaboratorsCount').textContent = 
            `${count} collaborator${count !== 1 ? 's' : ''}`;
    }

    broadcastCodeChange(newCode) {
        // This happens in real-time across all connected clients
        socket.emit('code_change', {
            roomId: 'demo-room',
            code: newCode,
            userId: currentUser.id
        });
    }
}

// Start collaborating!
const session = new CollaborationSession();
session.addParticipant({ id: 'user1', name: 'You' });
